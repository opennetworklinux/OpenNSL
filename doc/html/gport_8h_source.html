00027 #ifndef _SHR_GPORT_H
00028 #define _SHR_GPORT_H
00029 
00030 /*
00031  * Defines:
00032  *     _SHR_GPORT_*
00033  * Purpose:
00034  *     GPORT (Generic port) definitions. GPORT can be used to specify
00035  *     a {module, port} pair, trunk, and other port types.
00036  */
00037 
00038 #define _SHR_GPORT_NONE             (0)
00039 #define _SHR_GPORT_INVALID         (-1)
00040 #define _SHR_GPORT_TYPE_LOCAL               1  /* Port on local unit     */
00041 #define _SHR_GPORT_TYPE_MODPORT             2  /* Module ID and port     */
00042 #define _SHR_GPORT_TYPE_TRUNK               3  /* Trunk ID               */
00043 
00044 #define _SHR_GPORT_TYPE_SHIFT                           26
00045 #define _SHR_GPORT_TYPE_MASK                            0x3f
00046 
00047 #define _SHR_GPORT_MODID_SHIFT                          11
00048 #define _SHR_GPORT_MODID_MASK                           0x7fff
00049 #define _SHR_GPORT_PORT_SHIFT                           0
00050 #define _SHR_GPORT_PORT_MASK                            0x7ff
00051 #define _SHR_GPORT_TRUNK_SHIFT                          0
00052 #define _SHR_GPORT_TRUNK_MASK                           0x3ffffff
00053 
00054 #define _SHR_GPORT_IS_MODPORT(_gport)    \
00055         (((_gport) >> _SHR_GPORT_TYPE_SHIFT) == _SHR_GPORT_TYPE_MODPORT)
00056 
00057 #define _SHR_GPORT_MODPORT_SET(_gport, _module, _port)                       \
00058         ((_gport) = (_SHR_GPORT_TYPE_MODPORT   << _SHR_GPORT_TYPE_SHIFT)   | \
00059          (((_module) & _SHR_GPORT_MODID_MASK)  << _SHR_GPORT_MODID_SHIFT)  | \
00060          (((_port) & _SHR_GPORT_PORT_MASK)     << _SHR_GPORT_PORT_SHIFT))
00061 
00062 #define _SHR_GPORT_MODPORT_MODID_GET(_gport)    \
00063         (((_gport) >> _SHR_GPORT_MODID_SHIFT) & _SHR_GPORT_MODID_MASK)
00064 
00065 #define _SHR_GPORT_MODPORT_PORT_GET(_gport)     \
00066         (((_gport) >> _SHR_GPORT_PORT_SHIFT) & _SHR_GPORT_PORT_MASK)
00067 
00068 #endif  /* !_SHR_GPORT_H */
