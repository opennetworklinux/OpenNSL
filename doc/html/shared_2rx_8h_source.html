00379 /*
00380  * Macro to check if a reason (_SHR_RX_*) is included in a
00381  * set of reasons (_shr_rx_reasons_t). Returns:
00382  *   zero     => reason is not included in the set
00383  *   non-zero => reason is included in the set
00384  */
00385 #define _SHR_RX_REASON_GET(_reasons, _reason) \
00386         SHR_BITGET(((_reasons).pbits), (_reason))
00387 
00388 /*
00389  * Macro to add a reason (_SHR_RX_*) to a set of
00390  * reasons (_shr_rx_reasons_t)
00391  */
00392 #define _SHR_RX_REASON_SET(_reasons, _reason) \
00393         SHR_BITSET(((_reasons).pbits), (_reason))
00394 
00395 /*
00396  * Macro to add all reasons (_SHR_RX_*) to a set of
00397  * reasons (_shr_rx_reasons_t)
00398  */
00399 #define _SHR_RX_REASON_SET_ALL(_reasons) \
00400         SHR_BITSET_RANGE(((_reasons).pbits), 0, _SHR_RX_REASON_COUNT)

00402 /*

00404  * reasons (_shr_rx_reasons_t)
