00118 int opennsl_mac_parse(char *buf, unsigned char *macp)
00119 {
00120   int   i, c1, c2;
00121   char  *s;
00122 #define MAC_ADDR_LEN 17
00123 
00124   macp[0] = macp[1] = macp[2] = 0;
00125   macp[3] = macp[4] = macp[5] = 0;
00126 
00127   if ((buf == NULL) || (strlen(buf) > MAC_ADDR_LEN)) {
00128     return OPENNSL_E_FAIL;
00129   }
00130 
00131   /* skip leading 0x if plain hex format */
00132   if (buf[0] == '0' && (buf[1] == 'x' || buf[1] == 'X')) {
00133     buf += 2;
00134   }
00135 
00136   /* start at end of string and work backwards */
00137   for (s = buf; *s; s++) {
00138     ;
00139   }
00140   for (i = 5; i >= 0 && s >= buf; i--) {
00141     c1 = c2 = 0;
00142     if (--s >= buf) {
00143       if (*s >= '0' && *s <= '9') {
00144         c2 = *s - '0';
00145       } else if (*s >= 'a' && *s <= 'f') {
00146         c2 = *s - 'a' + 10;
00147       } else if (*s >= 'A' && *s <= 'F') {
00148         c2 = *s - 'A' + 10;
00149       } else if (*s == ':') {
00150         ;
00151       } else {
00152         return OPENNSL_E_FAIL;
00153       }
00154     }
00155     if (*s != ':' && --s >= buf) {
00156       if (*s >= '0' && *s <= '9') {
00157         c1 = *s - '0';
00158       } else if (*s >= 'a' && *s <= 'f') {
00159         c1 = *s - 'a' + 10;
00160       } else if (*s >= 'A' && *s <= 'F') {
00161         c1 = *s - 'A' + 10;
00162       } else if (*s == ':') {
00163         ;
00164       } else {
00165         return OPENNSL_E_FAIL;
00166       }
00167     }
00168 
00169     if (s > buf && s[-1] == ':') {
00170       --s;
00171     }
00172     macp[i] = c1 << 4 | c2;
00173   }
00174   return OPENNSL_E_NONE;
00175 }

00207 int opennsl_ip_parse(char *ip_str, unsigned int *ip_val)
00208 {
00209   unsigned int num = 0, val;
00210   char *tok;
00211   int count = 0;
00212   char buf[16]; /* Maximum length of IP address in dotten notation */
00213 
00214   if((ip_str == NULL) || (ip_val == NULL) || (strlen(ip_str) > 16))
00215   {
00216     return -1;
00217   }
00218   strcpy(buf, ip_str);
00219   tok = strtok(buf, ".");
00220   while(tok != NULL)
00221   {
00222     count++;
00223     val = atoi(tok);
00224     if((val < 0) || (val > 0xff))
00225     {
00226       return -1;
00227     }
00228     num = (num << 8) + val;
00229     tok = strtok(NULL, ".");
00230   }
00231   if(count != 4)
00232   {
00233     return -1;
00234   }
00235   *ip_val = num;
00236 
00237   return 0;
00238 }

00271 int opennsl_ctoi(const char *s, char **end)
00272 {
00273   unsigned int  n, neg;
00274   int base = 10;
00275 
00276   if (s == 0) {
00277     if (end != 0) {
00278       end = 0;
00279     }
00280     return 0;
00281   }
00282 
00283   s += (neg = (*s == '-'));
00284 
00285   if (*s == '0') {
00286     s++;
00287     if (*s == 'x' || *s == 'X') {
00288       base = 16;
00289       s++;
00290     } else if (*s == 'b' || *s == 'B') {
00291       base = 2;
00292       s++;
00293     } else {
00294       base = 8;
00295     }
00296   }
00297 
00298   for (n = 0; ((*s >= 'a' && *s < 'a' + base - 10) ||
00299         (*s >= 'A' && *s < 'A' + base - 10) ||
00300         (*s >= '0' && *s <= '9')); s++) {
00301     n = n * base + ((*s <= '9' ? *s : *s + 9) & 15);
00302   }
00303 
00304   if (end != 0) {
00305     *end = (char *) s;
00306   }
00307 
00308   return (int) (neg ? -n : n);
00309 }
